    Modify `middleware.ts` to add the setup completion check. The middleware already runs for all authenticated routes. Add the setup redirect logic:

    ```typescript
    import { prisma } from "@/lib/db"

    export default async function middleware(request: NextRequest) {
      // Existing auth check...
      if (globalConfig.selfHosted.isEnabled) {
        // In self-hosted mode, check setup completion
        const pathname = request.nextUrl.pathname

        // Skip redirect if already on /setup or on API/auth routes
        if (!pathname.startsWith("/setup") && !pathname.startsWith("/api")) {
          // Check if business profile is complete via a lightweight DB query
          // Query the settings table directly for the required biz_ codes
          const requiredCodes = ["biz_company_name", "biz_tax_id", "biz_branch", "biz_address"]
          const settings = await prisma.setting.findMany({
            where: {
              code: { in: requiredCodes },
              value: { not: "" },
            },
            select: { code: true },
          })
          const isComplete = requiredCodes.every(code =>
            settings.some(s => s.code === code)
          )
          if (!isComplete) {
            return NextResponse.redirect(new URL("/setup", request.url))
          }
        }
        return NextResponse.next()
      }

      // Cloud mode auth check (existing logic)...
    }
    ```

    NOTE: Import prisma from `@/lib/db`. The middleware runs on the Edge runtime by default in Next.js. If Prisma Client does not work in Edge middleware (common issue), use an alternative approach: set a cookie flag (`banchee_setup_complete=true`) after the setup wizard completes, and check that cookie in middleware instead of querying the DB. This is simpler and avoids Edge runtime limitations:

    // Cookie-based approach (preferred if Prisma doesn't work in middleware):
    const setupComplete = request.cookies.get("banchee_setup_complete")?.value
    if (!setupComplete && !pathname.startsWith("/setup") && !pathname.startsWith("/api")) {
      return NextResponse.redirect(new URL("/setup", request.url))
    }
